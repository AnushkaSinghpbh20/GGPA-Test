CREATE TABLE company_info (
  company_id INT(11) NOT NULL AUTO_INCREMENT,
  company_name VARCHAR(200) NOT NULL,
  website VARCHAR(255) DEFAULT NULL,
  logo VARCHAR(255) DEFAULT NULL,
  location VARCHAR(200) DEFAULT NULL,
  PRIMARY KEY (company_id)
);

CREATE TABLE training_tech (
  tech_id INT(11) NOT NULL AUTO_INCREMENT,
  technology_name VARCHAR(100) UNIQUE DEFAULT NULL,
  PRIMARY KEY (tech_id)
);

CREATE TABLE trainer_dts (
  trainer_id INT(11) NOT NULL AUTO_INCREMENT,
  name VARCHAR(100) UNIQUE DEFAULT NULL,
  PRIMARY KEY (trainer_id)
);

CREATE TABLE file_info (
  file_id INT(11) NOT NULL AUTO_INCREMENT,
  file_name VARCHAR(255) NOT NULL,
  session_year VARCHAR(10) NOT NULL,
  status ENUM('pending','done','error','deleted','moved') DEFAULT 'pending',
  uploaded_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (file_id)
);

CREATE TABLE training_dts (
  survey_id INT(11) NOT NULL AUTO_INCREMENT,
  file_id INT(11) DEFAULT NULL,
  timestamp DATETIME DEFAULT NULL,
  email_id VARCHAR(100) DEFAULT NULL,
  enroll_no VARCHAR(50) DEFAULT NULL,
  name VARCHAR(100) DEFAULT NULL,
  branch VARCHAR(50) DEFAULT NULL,
  session VARCHAR(50) DEFAULT NULL,
  company_id INT(11) DEFAULT NULL,
  project_name VARCHAR(100) DEFAULT NULL,
  start_dt DATE DEFAULT NULL,
  end_dt DATE DEFAULT NULL,
  location VARCHAR(100) DEFAULT NULL,
  fees DECIMAL(10,2) DEFAULT NULL,
  training_mode VARCHAR(50) DEFAULT NULL,
  overall_exp VARCHAR(200) DEFAULT NULL,
  remark TEXT DEFAULT NULL,
  PRIMARY KEY (survey_id),
  FOREIGN KEY (company_id) REFERENCES company_info(company_id),
  FOREIGN KEY (file_id) REFERENCES file_info(file_id) ON DELETE CASCADE
);

CREATE TABLE file_log (
  log_id INT(11) NOT NULL AUTO_INCREMENT,
  file_id INT(11) NOT NULL,
  session_year VARCHAR(10) NOT NULL,
  remarks TEXT DEFAULT NULL,
  log_time TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (log_id),
  FOREIGN KEY (file_id) REFERENCES file_info(file_id) ON DELETE CASCADE
);

CREATE TABLE comp_feedback (
  feedback_id INT(11) NOT NULL AUTO_INCREMENT,
  survey_id INT(11) DEFAULT NULL,
  company_id INT(11) DEFAULT NULL,
  communication VARCHAR(50) DEFAULT NULL,
  infra VARCHAR(50) DEFAULT NULL,
  activity VARCHAR(50) DEFAULT NULL,
  overall_fdb VARCHAR(50) DEFAULT NULL,
  remark TEXT DEFAULT NULL,
  proj_status VARCHAR(100) DEFAULT NULL,
  PRIMARY KEY (feedback_id),
  FOREIGN KEY (survey_id) REFERENCES training_dts(survey_id),
  FOREIGN KEY (company_id) REFERENCES company_info(company_id)
);

CREATE TABLE other_feedback (
  feedback_id INT(11) NOT NULL AUTO_INCREMENT,
  survey_id INT(11) DEFAULT NULL,
  location VARCHAR(100) DEFAULT NULL,
  accommodation VARCHAR(100) DEFAULT NULL,
  fooding VARCHAR(100) DEFAULT NULL,
  PRIMARY KEY (feedback_id),
  FOREIGN KEY (survey_id) REFERENCES training_dts(survey_id)
);

CREATE TABLE student_tech_map (
  id INT(11) NOT NULL AUTO_INCREMENT,
  survey_id INT(11) DEFAULT NULL,
  tech_id INT(11) DEFAULT NULL,
  PRIMARY KEY (id),
  FOREIGN KEY (survey_id) REFERENCES training_dts(survey_id),
  FOREIGN KEY (tech_id) REFERENCES training_tech(tech_id)
);

CREATE TABLE student_trainer_map (
  id INT(11) NOT NULL AUTO_INCREMENT,
  survey_id INT(11) DEFAULT NULL,
  trainer_id INT(11) DEFAULT NULL,
  PRIMARY KEY (id),
  FOREIGN KEY (survey_id) REFERENCES training_dts(survey_id),
  FOREIGN KEY (trainer_id) REFERENCES trainer_dts(trainer_id)
);

CREATE TABLE trainer_feedback (
  feedback_id INT(11) NOT NULL AUTO_INCREMENT,
  survey_id INT(11) DEFAULT NULL,
  trainer_id INT(11) DEFAULT NULL,
  company_id INT(11) DEFAULT NULL,
  teaching VARCHAR(50) DEFAULT NULL,
  knowledge VARCHAR(50) DEFAULT NULL,
  overall_fdb VARCHAR(50) DEFAULT NULL,
  remark TEXT DEFAULT NULL,
  PRIMARY KEY (feedback_id),
  FOREIGN KEY (survey_id) REFERENCES training_dts(survey_id),
  FOREIGN KEY (trainer_id) REFERENCES trainer_dts(trainer_id),
  FOREIGN KEY (company_id) REFERENCES company_info(company_id)
);

CREATE TABLE locations (
  id INT(11) NOT NULL AUTO_INCREMENT,
  name VARCHAR(100) NOT NULL,
  city VARCHAR(100) DEFAULT NULL,
  state VARCHAR(100) DEFAULT NULL,
  country VARCHAR(100) DEFAULT 'India',
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id)
);

CREATE TABLE staging_table (

  end_date VARCHAR(100) DEFAULT NULL,
  trainer_name VARCHAR(255) DEFAULT NULL,
  location VARCHAR(255) DEFAULT NULL,
  fees_submitted VARCHAR(100) DEFAULT NULL,
  training_mode VARCHAR(100) DEFAULT NULL,
  communication TEXT DEFAULT NULL,
  infrastructure TEXT DEFAULT NULL,
  extra_activities TEXT DEFAULT NULL,
  overall_company_exp VARCHAR(100) DEFAULT NULL,
  remark_company TEXT DEFAULT NULL,

  project_status VARCHAR(255) DEFAULT NULL,

  move_status ENUM('pending','moved','error') DEFAULT 'pending',

     CREATE TABLE session_ref(

CREATE TABLE survey_tracker (
  survey_id INT,
  enrollment_no VARCHAR(15) NOT NULL,
  session_value VARCHAR(30) NOT NULL,
  status VARCHAR(30) NOT NULL,
  
  PRIMARY KEY (survey_id),
